212 void lcd_setLine(int16_t x1, int16_t y1, int16_t x2, int16_t y2, uint8_t state)
213 {
214  int16_t i, x[2], y[2];
215  float xF, yF;
216 
217  xF = x1, x[1] = 0;
218  yF = y1, y[1] = 0;
219 
220  x[0] = abs(x2-x1);
221  y[0] = abs(y2-y1);
222 
223  if(x[0]) x[1] = (x2-x1)/x[0];
224  if(y[0]) y[1] = (y2-y1)/y[0];
225 
226  if(x[0] >= y[0]) {
227  for(i = 0; i<x[0]; i++) {
228  if( xF >= 0 && yF >= 0) lcd_setPixel(xF, yF, state);
229  xF += x[1];
230  yF += ((float)y[0]/x[0])*y[1];
231  }
232  }
233  else {
234  for(i = 0; i<y[0]; i++) {
235  if( xF >= 0 && yF >= 0) lcd_setPixel(xF, yF, state);
236  xF += ((float)x[0]/y[0])*x[1];
237  yF += y[1];
238  }
239  }
240 }
